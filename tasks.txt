Задача 7
Select product.model, price
FROM product, pc
where product.model = pc.model and maker= 'B'
union
Select product.model, price
FROM product, laptop
where product.model = laptop.model and maker= 'B'
union
Select product.model, price
FROM product, printer
where product.model = printer.model and maker= 'B'
